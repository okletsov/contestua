SELECT 
	case
        when `c`.`season` = 'Winter' then concat(substring(`c`.`season`, 1, 3), ' ', right(`c`.`year`, 2), '-', right(`c`.`year`, 2) + 1)
        else concat(substring(`c`.`season`, 1, 3), ' ', right(`c`.`year`, 2))
    end AS `label`
from contest c 
where 1=1
	and `type` = 'seasonal'
order by start_date desc 
limit 6;
